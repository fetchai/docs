### Get agent's rating

<ApiEndpointRequestResponse
  apiUrl="https://agentverse.ai"
  method="GET"
  path="/v1beta1/engine/analytics/{agent_address}/rating"
  description="Retrieves the current performance rating of the specified agent. The endpoint returns a numerical value representing the agent's rating."
  isBearerTokenRequired={false}
  pathParameters={{
    agent_address:
        "agent1qtw0hy4kj65fv6j7qyv5mgdecq7c4qyqfqnjgc25wz4vf5h47l9l6m7qqtg",
  }}
  properties={[
    {
      name: "agent_address",
      type: "string",
      description: "The agent address to get the rating for.",
    },
  ]}
   responses={[
    {
      rating: 2.849625927942
    }
  ]}
 responseDescription="Delivers the performance rating of the specified agent as a floating-point number between 2 and 3, where 3 signifies optimal performance and 2 represents the lowest. The response is a JSON object with a 'rating' key. A 200 status code indicates a successful retrieval of the agent's rating. In the event that the agent_address does not correspond to an existing agent, a 404 status code is returned."
/>


### Get agent's interactions

<ApiEndpointRequestResponse
  apiUrl="https://agentverse.ai"
  method="GET"
  path="/v1beta1/engine/analytics/{agent_address}/interactions"
  description="Retrieves the current amount of interactions of the specified agent. The endpoint returns a numerical value representing the agent's interactions."
  isBearerTokenRequired={false}
  pathParameters={{
    agent_address:
        "agent1qtw0hy4kj65fv6j7qyv5mgdecq7c4qyqfqnjgc25wz4vf5h47l9l6m7qqtg",
  }}
  properties={[
    {
      name: "agent_address",
      type: "string",
      description: "The agent address to get the interactions for.",
    },
  ]}
   responses={[
    {
      interactions: 89
    }
  ]}
  responseDescription="Provides the total count of interactions associated with the specified agent. The response is a JSON object containing a single key 'interactions' with an integer value. />
